// ==UserScript==
// @name         Askaban Chatbot ‚Äì Your Magical AI Wizard
// @namespace   
// @version      1.0
// @description  Injects Askaban chatbot trained on Sorcerer's Stone PDF
// @match      
// @grant        GM_addStyle
// ==/UserScript==

(function () {
    'use strict';

    // 1. Add Styles
    GM_addStyle(`
        #chat-widget {
            position: fixed;
            bottom: 20px;
            right: 20px;
            z-index: 9999;
            font-family: 'Segoe UI', sans-serif;
        }

        #chat-header {
            background: #6a0dad;
            color: white;
            padding: 10px 15px;
            border-radius: 16px 16px 0 0;
            font-weight: bold;
            display: flex;
            align-items: center;
        }

        #chat-header span {
            margin-left: 8px;
        }

        #chat-container {
            width: 500px;
            background: #1e1e1e;
            border-radius: 16px;
            overflow: hidden;
            box-shadow: 0 4px 10px rgba(0,0,0,0.3);
        }

        #chatbox {
            max-height: 400px;
            overflow-y: auto;
            padding: 10px;
            display: none;
        }

.user-message {
    background: #6a0dad;
    color: white;
    padding: 10px 14px;
    border-radius: 18px;
    max-width: 70%;
    width: auto;
    display: block;
    float: right;
    clear: both;
    margin: 10px 0 0 auto;
    text-align: right;
    word-wrap: break-word;
}

.bot-message {
    background: black;
    color: white;
    padding: 10px 14px;
    border-radius: 18px;
    max-width: 70%;
    width: auto;
    display: block;
    float: left;
    clear: both;
    margin: 10px auto 0 0;
    text-align: left;
    word-wrap: break-word;
}



        #chat-input-area {
            display: flex;
            padding: 10px;
            background: #2c2c2c;
        }

        #chat-input {
            flex: 1;
            padding: 10px;
            border: none;
            border-radius: 12px;
            background: #111;
            color: white;
            margin-right: 10px;
        }

        #chat-send {
            background: #6a0dad;
            color: white;
            border: none;
            border-radius: 12px;
            padding: 10px 16px;
            cursor: pointer;
        }

        #chat-send:hover {
            background: #5c09b0;
        }
    `);

    // 2. Add HTML
    const widget = document.createElement("div");
    widget.id = "chat-widget";
    widget.innerHTML = `
        <div id="chat-container">
            <div id="chat-header">
                üßô‚Äç‚ôÇÔ∏è <span>Meet Askaban ‚Äì Your Magical AI Wizard</span>
            </div>
            <div id="chatbox"></div>
            <div id="chat-input-area">
                <input type="text" id="chat-input" placeholder="Ask your question..." />
                <button id="chat-send">Send</button>
            </div>
        </div>
    `;
    document.body.appendChild(widget);

    // 3. Chat Logic
    const input = document.getElementById("chat-input");
    const sendBtn = document.getElementById("chat-send");
    const chatbox = document.getElementById("chatbox");

    function appendMessage(sender, message) {
        chatbox.style.display = "block";
        const msgDiv = document.createElement("div");
        msgDiv.className = sender === 'user' ? 'user-message' : 'bot-message';
        msgDiv.innerText = message;
        chatbox.appendChild(msgDiv);
        chatbox.scrollTop = chatbox.scrollHeight;
    }

    async function sendMessage() {
        const question = input.value.trim();
        if (!question) return;

        appendMessage('user', question);
        input.value = '';

        try {
            const res = await fetch("http://localhost:5000/query/ask", {
                method: "POST",
                headers: {
                    "Content-Type": "application/json"
                },
                body: JSON.stringify({ question: question })
            });

            const response = await res.json();
            appendMessage('bot-message', response.answer || '‚ö†Ô∏è No response.');
        } catch (err) {
            appendMessage('bot', "‚ö†Ô∏è Could not connect to Askaban server.");
        }
    }

    sendBtn.onclick = sendMessage;
    input.addEventListener("keydown", function (e) {
        if (e.key === "Enter") sendMessage();
    });

})();
